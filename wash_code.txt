def wear(stink)
  sample(:ambi_choir, rate: stink)
  sleep(1)
end

def wash(temperature)
  play(temperature)
  sleep(1)
end

def dry(heat)
  sample(heat)
  sleep(1)
end

3.times do
  wear(1)
  wash(70)
  dry(:elec_wood)
end

3.times do
  wear(4)
  wash(60)
  dry(:drum_cowbell)
end
